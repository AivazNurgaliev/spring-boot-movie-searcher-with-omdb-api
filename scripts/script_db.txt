create table public.account (
  account_id integer generated by default as identity primary key,
  email character varying(255) not null unique,
  password character varying(255) not null,
  username character varying(255) not null unique,
  role character varying(255) not null default 'USER',
  status character varying(255) default 'ACTIVE'
);

CREATE TABLE movie (
    movie_id integer GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    account_id integer REFERENCES account (account_id),
    title VARCHAR(255),
    year VARCHAR(255),
    imdbID VARCHAR(255),
    type VARCHAR(255),
    poster VARCHAR(255)
);
